		ret_param.startpos = 0;
		ret_param.startdistance = 640;
		tag.append(Tag_PosTyuuou);
		// tag.append(Tag_PosTyukyori);
		break;
	case "中央遠":
		ret_param.startpos = 0;
		ret_param.startdistance = 960;
		// tag.append(Tag_PosEnkyori);
		break;
	case "中央":
		tag.append(Tag_PosTyuuou);

		switch( cat )
		{
		// コンボならデフォが近く
		case Def_TCNum_BNBLv0:
		case Def_TCNum_BNBLv1:
		case Def_TCNum_BNBLv2:
		case Def_TCNum_BNBLv3:
		case Def_TCNum_Level0:
		case Def_TCNum_Level1:
		case Def_TCNum_Level2:
		case Def_TCNum_Level3:
		case Def_TCNum_Level20:
			ret_param.startpos = 0;
			ret_param.startdistance = 120;
			break;
		default:
			break;
		}
		break;
	}
	
	ret_param.startdistance = ("startdistance" in param)? param.startdistance : ret_param.startdistance; //i added this
	
	switch( gauge )
	{
	case "EX":
		ret_param.grdtype = 5; // 計測なし
		ret_param.exstype = 4; // 100%固定、コンボ切れで100%に戻る
		tag.append(Tag_GaugeEX);
		break;
	case "IW":
		ret_param.grdtype = 5; // 計測なし
		ret_param.exstype = 1; // MAX固定
		tag.append(Tag_GaugeIW);
		break;
	case "DCVO":
		ret_param.grdtype = 103; // 即時ヴォーパル GRD3個
		ret_param.exstype = 1; // MAX固定
		ret_param.hptype = 1; // 体力30％
		tag.append(Tag_GaugeDyingCVO);
		break;
	case "CVO":
		ret_param.grdtype = 103; // 即時ヴォーパル GRD3個
		ret_param.exstype = 1; // MAX固定
		tag.append(Tag_GaugeCVO);
		break;
	case "CS":
		ret_param.grdtype = 103; // 即時ヴォーパル GRD3個
		ret_param.exstype = 2; // ZERO
		tag.append(Tag_GaugeCS);
		break;
	case "EXCS":
		ret_param.grdtype = 103; // 即時ヴォーパル GRD3個
		ret_param.exstype = 4; // ZERO
		tag.append(Tag_GaugeEXCS);
		break;

		ret_param.grdtype = 5; // 計測なし
		ret_param.hptype = 1; // 瀕死
		ret_param.exstype = 1; // MAX固定
		tag.append(Tag_GaugeIWE);
		break;

		ret_param.grdtype = 5; // 計測なし
		ret_param.exstype = 2; // ZERO
		// tag.append(Tag_GaugeNone);
		break;

	switch( enemy )
	{
	case "屈":
		ret_param.enemyguard = 2; // 状態ガード
		ret_param.enemytype = 2;
		tag.append(Tag_EnemyOverhead);
		break;
	case "ジャンプ":
		ret_param.enemytype = 1;
		tag.append(Tag_EnemyAntiAir);
		break;

		ret_param.enemytype = 255;
		break;
	case "COM":
		ret_param.enemytype = 19;
		break;
	case "立":
		ret_param.enemyguard = 2; // 状態ガード
		tag.append(Tag_EnemyLow);
		break;
	case "ガード":
		ret_param.enemyguard = 1; // ALLガード
		break;
	case "COM立": 
		ret_param.enemytype = 19;
		tag.append(Tag_EnemyLow); // COMで下段のタグ付けたい時に
		break;

		ret_param.enemytype = ("enemytype" in param)? param.enemytype : 0;
		ret_param.enemyguard = ("enemyguard" in param)? param.enemyguard : 0;

	local tc_combo = 0;
	
	switch( cat )
	{
	case Def_TCNum_BNBLv0:
	case Def_TCNum_BNBLv1:
	case Def_TCNum_BNBLv2:
	case Def_TCNum_BNBLv3:
	case Def_TCNum_Level0:

	case Def_TCNum_Level2:

	case Def_TCNum_Level20:

		break;

	case Def_TCNum_Level20:
	case Def_TCNum_TacticsLv3:

			sidou_tag = "("+sidouSkill+"〜"+simeSkill+")";

		}

			if( tmp_str != "*" )
			{
				end = tmp_str;
			}			
		}		
	
		if( expl )
		{
			local tmp_path = tmp_chr_header+"_tag_explanation_"+tmp_skill_num_str;
			local tmp_str = sqfunc.GetExStringText(tmp_path);
			if( tmp_str != "*" )
			{
				expl = tmp_str;
			}
		}
		if( poin || step )
		{
			local tmp_path = tmp_chr_header+"_tag_hint_"+tmp_skill_num_str;
			local tmp_str = sqfunc.GetExStringText(tmp_path);
			if( tmp_str != "*" )
			{
				// print("\n ヒント定義あったっぽいので:"+num+"\n"+tmp_str);
				csvHint = tmp_str;
			}
		}
	}

	// 挿入タイプ
	// チュートリアルみたいな文章
	if( start )
	{
		ret_param.start <- start;
	}
			
	if( altstart )
	{
		ret_param.start <- altstart;
	}
	
	if( end )
	{
		ret_param.end <- end;
	}
	
	local hint_mess = "";
	
	if( hint )
	{
		hint_mess = hint;
	}
	else
	{
		if( expl )
		{
			hint_mess += Hint_Info+"\\n"+expl+"\\n\\n";
		}
		
		// csvの定義はpoinとstepをまとめて書いてあるので
		// 分岐を追加
		if( csvHint )
		{
			hint_mess += csvHint;//\n\nはcsvにあるので不要
		}
		else
		{
			if( poin )
			{
				hint_mess += Hint_Point+"\\n"+poin+"\\n\\n";
			}
			if( step )
			{
				hint_mess += Hint_Step+"\\n"+step+"\\n\\n";
			}
		}
		if( 1 ) // 変更履歴書き出しのときだけ0に
		{
			hint_mess += Hint_Icon+"\\n"+Hint_IconInfo+"\\n\\n";
		}
	}
	if( hint_mess != "" )
	{
		ret_param.hint <- hint_mess;
	}
	
	if( expl )
	{
		ret_param.explanation <- expl;
	}
	else
	{
		// ret_param.explanation <- "説明はありません。";
	}
	
	if( player )
	{
		ret_param.player <- player;
	}
	
	if( stringID )
	{
		ret_param.stringID <- stringID;
		
		// デバッグ処理XXX
		if( 0 )
		{
			//ファイルの存在チェック
			//aka_Lv1_poke01
			//aka_Lv1_poke01mvp.dat
			local file = stringID + "mvp.dat";
			ret_param.explanation = "【"+stringID + "】" + ret_param.explanation;
			
			// 大文字チェック
			local re = regexp(@"[A-Z]+"); 
			local firstMatch = re.search(stringID);
			if( firstMatch )
			{
				ret_param.title = "【stringID error】"+ret_param.title;
			}
		}
	}
	
	// skilltypeの指定があったらカテゴリ問わず上書き
	if( "skilltype" in param )
	{
		ret_param.skilltype = skilltype;
	}

	// endtypeの指定があったらカテゴリ問わず上書き
	if( "endtype" in param )
	{
		ret_param.endtype = endtype;
	}
	
	// viewtypeの指定があったらカテゴリ問わず上書き
	if( "viewtype" in param )
	{
		ret_param.viewtype = viewtype;
	}
	
	return ret_param;
}

local margeTableVal = function( _src, _dst )
{
	local ret = clone _dst;
	// srcの内容をdstに上書きしていく
	
	foreach( slot, val in _src )
	{
		// print("\n slot:"+slot+" v:"+val );
		if( slot in ret )
		{
			// print(" -> set" );
			// resetが元々記述してあるものは上書きしない（攻撃オブジェクトが別なやつとか移動技なので）
			if( slot != "reset")
			{
				ret[slot] = val;
			}
		}
		else
		{
			// print(" -> add" );
			ret[slot] <- val;
		}	
	}
	return ret;
}

local getSkillEff = function( wordparam ) : (margeTableVal)
{
	// print("\n atkword:"+atkword );
	
	local ret_param =
	{
		word = "",
		param = {}, // この値がskillListに上書きされる
		addtext = "", // 後
		instext = "", // 手前
		ret = 0, // 改行？
	};	
	
	local atkword = "";
	local addparam = 0;
	if( typeof wordparam == "table" )
	{
		if( "skill" in wordparam )
		{
			atkword = wordparam.skill;
		}
		if( "add" in wordparam )
		{
			addparam = wordparam.add;
		}
	}
	else
	{
		// 文字列のときだと思う
		atkword = wordparam;
	}
	
	ret_param.word = atkword;

	
	/*
	foreach( slot, val in ret_param.param )
	{
		print("\n ★slot:"+slot );
	}
	*/
	
	// コンボは横書き
	// ret_param.param.postype <- 1; // 横にかく
	
	if( ret_param.word.find("(")==0 && ret_param.word.find(")")==ret_param.word.len()-1 )
	{
		ret_param.word = ret_param.word.slice(1,ret_param.word.len()-1 );
		ret_param.param.reset <- 2;
		ret_param.addtext += " "+WKaraburi;
	}
	
	// ディレイをかける
	if( ret_param.word.find("d")==0 )
	{
		ret_param.word = ret_param.word.slice(1); // >以降を追加
		ret_param.instext += "\\@V@";
	}
	
	// 着地後に技を出す
	if( ret_param.word.find("l")==0 && ret_param.word.find("lin")!=0 )
	{
		ret_param.word = ret_param.word.slice(1); // >以降を追加
		ret_param.instext = WAfterLand + ret_param.instext;
	}
	

	// インクリース技のチェック
	if( ret_param.word.find("i")==0 )
	{
		
		ret_param.word = ret_param.word.slice(1); // i以降を追加
		// ret_param.addtext =+= WIC;
		ret_param.instext += WIC;
		ret_param.param.mvcode1 <- def_MC1_ExtendAction; // インクリース

		// ret_param.param.type <- 1; // Flag監視タイプ
		// ret_param.param.mvflag <- 4; // インクリース
		// print("\n atkword3:"+ret_param.word );
	}
	
	// ボタンホールド
	if( ret_param.word.find("h")==0 )
	{
		
		ret_param.word = ret_param.word.slice(1); // i以降を追加
		ret_param.addtext += "\\@W@";
	}
	
	// print("\n ret_param.word:"+ret_param.word );
	// 派生技
	if( ret_param.word.find(">")==0 )
	{
		
		ret_param.word = ret_param.word.slice(1); // >以降を追加
		ret_param.instext += WAdd;//;
		// print("\n 追加:"+ret_param.word );
	}
	
	
	// print("\n ret_param.word:"+ret_param.word );
	// 状態監視からの技
	if( ret_param.word.find("*")==0 )
	{
		
		ret_param.word = ret_param.word.slice(1); // >以降を追加
		ret_param.param.reset <- 3; // リセット書き換え
	}
	
	// print("\n ret_param.word:"+ret_param.word );
	// 設置技の設置
	if( ret_param.word.find("_")==0 )
	{
		
		ret_param.word = ret_param.word.slice(1); // >以降を追加
		ret_param.param.reset <- 2; // リセット書き換え
		ret_param.addtext += WSetting;
	}
	
	// print("\n ret_param.word:"+ret_param.word );
	// 設置技のヒット
	if( ret_param.word.find("=")==0 )
	{
		
		ret_param.word = ret_param.word.slice(1); // >以降を追加
		ret_param.addtext += WSetHit;
	}	
	
	// print("\n ret_param.word:"+ret_param.word );
	// 隙消し（戦術指南で使用）
	if( ret_param.word.find("f")==0 )
	{
		
		ret_param.word = ret_param.word.slice(1); // >以降を追加
		ret_param.param.reset <- 4;// ヒットガードを問わない
		ret_param.addtext += WDelayClear;
	}
	
	// カウンター始動
	if( ret_param.word.find("c")==0 && ret_param.word.find("car")!=0 && ret_param.word.find("cha")!=0 && ret_param.word.find("checkpoint")!=0 )
	{	
		ret_param.word = ret_param.word.slice(1); // >以降を追加
		ret_param.param.mvcode2 <- def_MC2_CounterHit;
		ret_param.addtext += WCounterHit;
	}
	
	// print("\n ret_param.word:"+ret_param.word );
	// 改行テスト
	if( ret_param.word.find("r")==0 )
	{
		ret_param.word = ret_param.word.slice(1); // >以降を追加
		ret_param.param.postype <- 0; // 改行
		ret_param.ret = 1;
	}
	
	// ヒット数のチェック
	//(2)で2ヒットするまでチェックになる
	{
		local k_st = ret_param.word.find("(");
		local k_ed = ret_param.word.find(")");
		if( k_st !=-1 && k_ed !=-1 && k_st < k_ed && k_st>0 )
		{
			local word = ret_param.word;
			ret_param.word = word.slice(0,k_st); // (より前を追加
			
			// print("\n BB:"+ret_param.word.len() );
			local hitcount = word.slice(k_st+1,k_ed);
			local iHitcount = 0;
			// print("\n typeof hitcount :"+typeof hitcount+" hitcount:"+hitcount  );
			if( typeof hitcount =="string" )
			{
				iHitcount = hitcount.tointeger();
			}
			
			if( iHitcount > 0 )
			{
				// print("\n hit:"+iHitcount);
				ret_param.addtext += " "+iHitcount+WHitCount;
				// ret_param.param.time <- iHitcount-1; // time==0で既に1ヒット、1で2ヒット分
				ret_param.param.time <- iHitcount; // time==0でも1ヒットが検出されるが、1以降はヒット数と等しい
			}
			// print("\n atkword3:"+ret_param.word );
			
			// print("\n ヒット数:"+ret_param.word );
		}
	}
	
	//[2]で2ヒットするまでチェックになる(ヒット数非表示)
	{
		local k_st = ret_param.word.find("[");
		local k_ed = ret_param.word.find("]");
		if( k_st !=-1 && k_ed !=-1 && k_st < k_ed && k_st>0 )
		{
			local word = ret_param.word;
			ret_param.word = word.slice(0,k_st); // (より前を追加
			
			// print("\n BB:"+ret_param.word.len() );
			local hitcount = word.slice(k_st+1,k_ed);
			local iHitcount = 0;
			// print("\n typeof hitcount :"+typeof hitcount+" hitcount:"+hitcount  );
			if( typeof hitcount =="string" )
			{
				iHitcount = hitcount.tointeger();
			}
			
			if( iHitcount > 0 )
			{
				// print("\n hit:"+iHitcount);
				// ret_param.addtext += " "+iHitcount+"ヒット";
				// ret_param.param.time <- iHitcount-1; // time==0で既に1ヒット、1で2ヒット分
				ret_param.param.time <- iHitcount; // time==0でも1ヒットが検出されるが、1以降はヒット数と等しい
			}
			// print("\n atkword3:"+ret_param.word );
			
			// print("\n ヒット数:"+ret_param.word );
		}
	}
	// {B}でずらしB離す（主にカーマインのずらしホールド用）
	{
		local k_st = ret_param.word.find("{");
		local k_ed = ret_param.word.find("}");
		if( k_st !=-1 && k_ed !=-1 && k_st < k_ed && k_st>0 )
		{
			local word = ret_param.word;
			ret_param.word = word.slice(0,k_st); // (より前を追加
			
			local ReleaseButton = word.slice(k_st+1,k_ed);
			if( ReleaseButton )
			{
				ret_param.addtext += WZurasi+"\\@"+ReleaseButton+"@"+WZurasiRelease;
			}
		}
	}
	
	//WAllBtnRelease
	if( ret_param.word.find("#A#")==0 )
	{
		ret_param.word = ret_param.word.slice(3); // #_#以降を追加
		ret_param.addtext += WABtnRelease;
	}
	//WAllBtnRelease
	if( ret_param.word.find("#B#")==0 )
	{
		ret_param.word = ret_param.word.slice(3); // #_#以降を追加
		ret_param.addtext += WBBtnRelease;
	}
	//WAllBtnRelease
	if( ret_param.word.find("#C#")==0 )
	{
		ret_param.word = ret_param.word.slice(3); // #_#以降を追加
		ret_param.addtext += WCBtnRelease;
	}
	
	// 状態の一致でOK
	if( ret_param.word.find(":")==0 )
	{
		ret_param.word = ret_param.word.slice(1); // >以降を追加
		ret_param.param.reset <- 4; // リセット書き換え
		ret_param.instext += "";
	}

	// 最終的にparamにさらにaddを上書きする
	if( addparam )
	{
		ret_param.param = margeTableVal( ret_param.param, addparam );
	}
	
	return ret_param;
}

		if( sef.ret )

			dmgRecipeLength = 0;
		}
		
		word = sef.word;
		
		// ジャンプキャンセル抜けを探す
		/*
		// 2C > JB ×
		
		if( word.find("J") == 0 && dbgLastWord != "" && cat != Def_TCNum_TacticsLv1 )
